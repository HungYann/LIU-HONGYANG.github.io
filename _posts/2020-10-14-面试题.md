```java
public class Code {
    public static void main(String[] args) {
        Code code = new Code();
        code.decimalToOctal(10);
    }

    public void decimalToOctal(int k){
        int d, s = 0, c = 0;


        while (k != 0) {
            d = k % 8;
            s += d * (int) Math.pow(10, c++);
            k /= 8;
        }

        System.out.println("Octal equivalent:" + s);
    }

}
```
